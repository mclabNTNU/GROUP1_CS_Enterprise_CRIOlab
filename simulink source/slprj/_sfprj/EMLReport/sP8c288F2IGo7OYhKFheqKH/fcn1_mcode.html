1function [a_1, tau] = fcn(z1,z2,s_dot,eta_v,nu)
 2%Kinetics
 3u   = nu(1);
 4v   = nu(2);
 5r   = nu(3);
 6% MATRICES
 7m    = 14.79;   
 8I_z  =  1.76;  
 9x_g = 0.0375;       
10% Added mass
11X_ud  =     -2.0;                       % Table B.1
12Y_vd =     -1.0;    N_vd =    -0.0;     % Table B.1 guestimate
13Y_rd =      -0.0;   N_rd =    -1.0;     % Table B.1
14
15% Total mass matrix
16m_11 = m-X_ud;
17m_22 = m-Y_vd;
18m_23 = m*x_g-Y_rd;
19m_32 = m*x_g-N_vd;
20m_33 = I_z-N_rd;
21
22M    = [m_11 0    0
23        0    m_22 m_23
24        0    m_32 m_33];
25    
26% Damping coefficients
27% Comments at the end of the lines indicate the old values used.
28X_u     = -0.6555;      X_v     = 0;
29X_uu    = 0.3545;       X_vv    = -2.443;
30X_uuu   = -3.787;       X_vvv   = 0;
31Y_v     = -1.33;        Y_r     = -7.250; % Y_r
32Y_vv    = -2.776;       Y_rr    = -3.450; % Y_rr
33Y_vvv   = -64.91;       Y_rrr   = 0;
34Y_rv    = -0.805;       Y_vr    = -0.845; % Y_rv, Y_vr
35N_v     = 0;            N_r     = -1.900; % N_r
36N_vv    = -0.2088;      N_rr    = -0.750; % N_rr
37N_vvv   = 0;            N_rrr   = 0;
38N_rv    = 0.130;        N_vr    = 0.080; % N_rv, N_vr
39
40% Correolis matrix
41c_13 = -(m-Y_vd)*v-(m*x_g-Y_rd)*r;
42c_23 = (m-X_ud)*u;
43C    = [0     0     c_13
44        0     0     c_23
45        -c_13 -c_23 0   ];
46
47% Assembly of the damping matrix
48d_11 = -X_u - X_uu*u - X_uuu*u^2;
49d_22 = -Y_v - Y_vv*v - Y_vvv*v^2 - Y_rv*abs(r);
50d_33 = -N_r - N_rr*abs(r) - N_rrr*r^2 - N_vr*abs(v); % using abs(r)
51
52d_12 = -X_v - X_vv*v - X_vvv*v^2;
53d_23 = -Y_r - Y_rr*abs(r) - Y_rrr*r^2 - Y_vr*abs(v);
54d_32 = -N_v - N_vv*v - N_vvv*v^2 - N_rv*abs(r);
55
56D    = [d_11 d_12 0
57        0    d_22 d_23
58        0    d_32 d_33];
59
60    
61
62%Eta_d derivatives and U_s
63eta     = eta_v(1:3);
64eta_ds  = eta_v(4:6);
65eta_dss = eta_v(7:9);
66U_s     = eta_v(10);
67U_ss    = eta_v(11);
68%Rotation matrix and derivative
69R = [cos(eta(3)) -sin(eta(3)) 0
70     sin(eta(3))  cos(eta(3)) 0
71     0            0           1];
72 
73S = [0     -eta(3) 0
74     eta(3)   0    0
75     0        0    0]; 
76 
77%Gain matrices 
78Kp = diag([0.2,0.2,0.1]);
79Kd = diag([1, 1, 1]);
80
81%Controller alpha 1
82a_1 = -Kp*z1 +R'*eta_ds*U_s;
83
84
85%Alpha derivative
86a_t = -Kp*(-S*z1+nu)-S*R'*eta_ds*U_s;
87a_s = -Kp*R'*eta_ds+R'*eta_dss*U_s+R'*eta_ds*U_ss;
88a_dot = a_t + a_s*s_dot;
89
90%Controller tau
91tau = D*nu + C*nu -z1 +M*a_dot-Kd*z2;
92%tau = [tau(1);tau(2);tau(3)];
